public class Login{

    public void start(){
        Stage window = new Stage();
        window.setTitle("Авторизация");
        window.initModality(Modality.APPLICATION_MODAL);  // нельзя юзать другие окна
        window.setResizable(false);

        Button login = new Button("Войти на свой аккаунт");
        Button registration = new Button("Создать новый аккаунт");

        login.setOnAction(e -> {
            window.close();
        });
        registration.setOnAction(e -> {
            window.close();
        });

        VBox vBox = new VBox();
        vBox.getChildren().addAll(login, registration);
        Scene scene = new Scene(vBox, 270, 160);
        window.setScene(scene);
        window.show();
    }
}
